Startup screen showing LiterativeOS system status, recent files, and quick actions.

** Init

#+BEGIN_SRC elisp :init
(setq dashboard-startup-banner 'logo
      dashboard-center-content t
      dashboard-vertically-center-content t
      dashboard-show-shortcuts nil
      dashboard-items '((recents   . 10)
                        (bookmarks . 5)
                        (projects  . 5)))

;; Custom banner title
(setq dashboard-banner-logo-title
      (concat "🏗️  LITERATIVEOS - " (user-login-name) "@" (system-name)))

;; Set width for better visibility
(setq dashboard-item-shortcuts nil)
(setq dashboard-startupify-list
      '(dashboard-insert-banner
        dashboard-insert-newline
        dashboard-insert-banner-title
        dashboard-insert-newline
        dashboard-insert-navigator
        dashboard-insert-newline
        dashboard-insert-items
        dashboard-insert-newline
        dashboard-insert-footer))
#+END_SRC

** Config

#+BEGIN_SRC elisp :config
(dashboard-setup-startup-hook)

;; Custom format function for recents - show full path
(defun dashboard-insert-recents (list-size)
  "Insert recent files with full paths."
  (dashboard-insert-section
   "Recent Files:"
   (mapcar (lambda (file)
             (list (abbreviate-file-name file)))
           (take list-size recentf-list))
   list-size
   "r"
   `(lambda (&rest _)
      (find-file (car items)))))

;; Add system info section
(add-to-list 'dashboard-item-generators
  '(system . dashboard-insert-system-info) t)

(defun dashboard-insert-system-info (list-size)
  "Insert LiterativeOS system information."
  (let ((uptime (string-trim (shell-command-to-string "uptime -p")))
        (memory (car (split-string (string-trim (shell-command-to-string "free -h | grep Mem | awk '{print $3 \" / \" $2}'")) "\n")))
        (disk (car (split-string (string-trim (shell-command-to-string "df -h / | tail -1 | awk '{print $5}'")) "\n"))))
    (insert "\n")
    (insert (propertize "📊 System Status\n" 'face 'dashboard-heading))
    (insert (format "   Uptime:  %s\n" uptime))
    (insert (format "   Memory:  %s\n" memory))
    (insert (format "   Disk:    %s used\n" disk))
    (insert "\n")
    (insert (propertize "🤖 Emacs & AI\n" 'face 'dashboard-heading))
    (insert (format "   Packages: %d loaded\n" 
                    (length (list (featurep 'avy) (featurep 'god-mode) 
                                 (featurep 'opencode) (featurep 'org-mcp)))))
    (insert (format "   OpenCode: %s\n" (if (featurep 'opencode) "✓ Ready" "✗ Not loaded")))
    (insert (format "   org-mcp:  %s\n" (if (featurep 'org-mcp) "✓ Ready" "✗ Not loaded")))
    (insert "\n")
    (insert (propertize "⚡ Quick Actions\n" 'face 'dashboard-heading))
    (insert "   C-c o s  - OpenCode session\n")
    (insert "   C-c o m  - Enable MCP server\n")
    (insert "   C-x g    - Magit status\n")
    (insert "   C-x p p  - Project\n")))

;; Add system to dashboard items
(setq dashboard-items (append dashboard-items '((system . 1))))
#+END_SRC
