** Dired — The Filesystem Blade :config:
#+begin_src emacs-lisp
;; ════════════════════════════════════════════════════════════
;; § DIRED — DVORAK-OPTIMIZED NAVIGATION & POWER OPERATIONS
;; ════════════════════════════════════════════════════════════
(with-eval-after-load 'dired
  (evil-define-key 'normal dired-mode-map
    ;; ────────────────────────────────────────────────────────
    ;; Core Navigation (Dvorak HTNS)
    ;; ────────────────────────────────────────────────────────
    (kbd "h")         #'dired-up-directory              ; left: parent dir
    (kbd "s")         #'dired-find-file                 ; right: enter/open
    (kbd "t")         (lambda () 
                        (interactive)
                        (dired-next-line 1)
                        (when (bound-and-true-p shapeshifter/dired-preview-window)
                          (shapeshifter/dired-preview-file)))  ; down + auto-preview
    (kbd "n")         (lambda () 
                        (interactive)
                        (dired-previous-line 1)
                        (when (bound-and-true-p shapeshifter/dired-preview-window)
                          (shapeshifter/dired-preview-file)))  ; up + auto-preview
    
    ;; ────────────────────────────────────────────────────────
    ;; Enhanced Actions
    ;; ────────────────────────────────────────────────────────
    (kbd "SPC")       #'dired-find-file
    (kbd "RET")       #'dired-find-file
    (kbd "TAB")       (lambda () 
                        (interactive)
                        (if (file-directory-p (dired-get-filename nil t))
                            (dired-find-file)
                          (message "Use 'p' for preview if available")))
    (kbd "<tab>")     (lambda () 
                        (interactive)
                        (if (file-directory-p (dired-get-filename nil t))
                            (dired-find-file)
                          (message "Use 'p' for preview if available")))
    (kbd "p")         #'ignore  ; placeholder for preview
    (kbd "q")         #'quit-window
    (kbd "Q")         #'quit-window                      ; quit dired
    
    ;; ────────────────────────────────────────────────────────
    ;; Arrow keys (for optional use with auto-preview)
    ;; ────────────────────────────────────────────────────────
    (kbd "<left>")    #'dired-up-directory
    (kbd "<right>")   #'dired-find-file
    (kbd "<up>")      (lambda () 
                        (interactive)
                        (dired-previous-line 1)
                        (when (bound-and-true-p shapeshifter/dired-preview-window)
                          (shapeshifter/dired-preview-file)))
    (kbd "<down>")    (lambda () 
                        (interactive)
                        (dired-next-line 1)
                        (when (bound-and-true-p shapeshifter/dired-preview-window)
                          (shapeshifter/dired-preview-file)))
    
    ;; ────────────────────────────────────────────────────────
    ;; Sorting (z prefix - "zort") - add your own functions
    ;; ────────────────────────────────────────────────────────
    (kbd "zn")        (lambda () (interactive) (dired-sort-toggle-or-edit))
    (kbd "zt")        (lambda () (interactive) (dired-sort-toggle-or-edit))
    
    ;; ────────────────────────────────────────────────────────
    ;; Metadata & Analysis (g prefix) - add your own functions
    ;; ────────────────────────────────────────────────────────
    (kbd "gi")        #'dired-show-file-type
    
    ;; ────────────────────────────────────────────────────────
    ;; File Operations (vim-style mnemonics)
    ;; ────────────────────────────────────────────────────────
    (kbd "yy")        #'dired-copy-filename-as-kill     ; yank filename
    (kbd "Y")         (lambda () 
                        (interactive)
                        (kill-new (dired-get-filename)))  ; yank full path
    (kbd "d")         #'dired-do-delete                 ; delete (fixed!)
    (kbd "x")         #'dired-do-flagged-delete         ; execute flagged deletes
    (kbd "r")         #'dired-do-rename                 ; rename
    (kbd "c")         #'dired-do-copy                   ; copy
    (kbd "C")         #'dired-do-compress-to            ; compress
    (kbd "R")         #'dired-do-rename  ; bulk rename fallback
    (kbd "m")         #'dired-mark                      ; mark
    (kbd "u")         #'dired-unmark                    ; unmark
    (kbd "U")         #'dired-unmark-all-marks          ; unmark all
    (kbd "*")         #'dired-mark-executables          ; mark executables
    (kbd "v")         #'dired-mark                      ; visual mark (evil-like)
    
    ;; ────────────────────────────────────────────────────────
    ;; Buffer & Window Management
    ;; ────────────────────────────────────────────────────────
    (kbd "K")         #'kill-buffer
    (kbd "gr")        #'revert-buffer
    (kbd "C-r")       #'revert-buffer
    
    ;; ────────────────────────────────────────────────────────
    ;; Advanced Operations
    ;; ────────────────────────────────────────────────────────
    (kbd "X")         #'dired-do-flagged-delete
    (kbd "!")         #'dired-do-shell-command
    (kbd "&")         #'dired-do-async-shell-command
    (kbd "+")         #'dired-create-directory
    (kbd "=")         #'dired-diff
    
    ;; ────────────────────────────────────────────────────────
    ;; Toggle Features
    ;; ────────────────────────────────────────────────────────
    (kbd ".")         (lambda () 
                        (interactive)
                        (dired-hide-details-mode 'toggle))
    (kbd "i")         #'dired-toggle-read-only          ; edit mode
    
    ;; ────────────────────────────────────────────────────────
    ;; Jump & Search
    ;; ────────────────────────────────────────────────────────
    (kbd "/")         #'evil-search-forward
    (kbd "?")         #'evil-search-backward
    (kbd "gg")        #'evil-goto-first-line
    (kbd "G")         #'evil-goto-line)
  
  ;; ────────────────────────────────────────────────────────
  ;; Visual State — Mass Operations
  ;; ────────────────────────────────────────────────────────
  (evil-define-key 'visual dired-mode-map
    (kbd "d")         #'dired-do-delete
    (kbd "c")         #'dired-do-copy
    (kbd "r")         #'dired-do-rename
    (kbd "!")         #'dired-do-shell-command
    (kbd "C")         #'dired-do-compress-to))
#+end_src
