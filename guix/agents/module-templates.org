** Template A — Thin submodule (package exists in Guix)

#+BEGIN_SRC scheme
(define-module (core-system user-space root <CATEGORY> <NAME>)
  #:use-module (gnu packages <GNU-MODULE>)
  #:re-export (<NAME>))
#+END_SRC

Find the correct (gnu packages ...) module:
#+BEGIN_SRC sh
guix search <name>
# look at: location: gnu/packages/<module>.scm
#+END_SRC
Or in REPL: ,apropos <name>

Real example:
#+BEGIN_SRC scheme
(define-module (core-system user-space root shell nushell)
  #:use-module (gnu packages nushell)
  #:re-export (nushell))
#+END_SRC

#+END_SRC

*** npm fixed-output derivation pattern

For npm packages where the tarball URL is unstable, use fixed-output derivation:

#+BEGIN_SRC scheme
(source
 (origin
   (method url-fetch)
   (uri "https://registry.npmjs.org/<package>/-/<package>-<version>.tgz")
   (sha256
    (base32 "<hash>"))
   (snippet #~(delete-file-recursively "node_modules"))))  ; remove any bundled deps
#+END_SRC

For packages with bundled dependencies, use copy-build-system:

#+BEGIN_SRC scheme
(define-public <NAME>
  (package
    (name "<name>")
    (version "<version>")
    (source (origin
              (method url-fetch)
              (uri "<url>")
              (sha256 (base32 "<hash>"))))
    (build-system copy-build-system)
    (arguments
     '(#:install-plan
       '(("dist/" "bin/"))))  ; adjust paths as needed
    (synopsis "<short description>")
    (description "<longer description>")
    (home-page "<url>")
    (license license:<license>)))
#+END_SRC

*** Pre-built binary pattern (copy-build-system)

For packages with pre-built Linux binaries (GitHub releases):

#+BEGIN_SRC scheme
(define-public <NAME>
  (package
    (name "<name>")
    (version "<version>")
    (source (origin
              (method url-fetch)
              (uri (string-append "https://github.com/<user>/<repo>/releases/download/v" version "/<binary>-<version>-linux-amd64.tar.gz"))
              (sha256
               (base32 "<hash-from-guix-download>"))))
    (build-system copy-build-system)
    (arguments
     '(#:install-plan
       '(("<binary>" "bin/"))))
    (synopsis "<short description>")
    (description "<longer description>")
    (home-page "<url>")
    (license license:expat)))
#+END_SRC

Get hash: `guix download <url>` — copy the base32 string directly.

*** Pre-built binary with dynamic linker fix (patchelf)

If the pre-built binary is dynamically linked (check with `file <binary>`), it will fail RUNPATH validation on Guix. Fix with patchelf:

#+BEGIN_SRC scheme
(define-module (core-system user-space root <category> <name>)
  #:use-module (guix packages)
  #:use-module (guix download)
  #:use-module (guix build-system copy)
  #:use-module (guix utils)
  #:use-module ((guix licenses) #:prefix license:)
  #:use-module (gnu packages elf)
  #:use-module (gnu packages gcc)
  #:export (<name>))

(define-public <name>
  (package
    (name "<name>")
    (version "<version>")
    (source (origin
              (method url-fetch)
              (uri (string-append "https://github.com/<user>/<repo>/releases/download/v" version "/<binary>-<version>-linux-amd64.tar.gz"))
              (sha256
               (base32 "<hash>"))))
    (build-system copy-build-system)
    (native-inputs (list patchelf))
    (inputs (list glibc (list gcc "lib")))
    (arguments
     '(#:install-plan
       '(("<binary>" "bin/"))
       #:validate-runpath? #f
       #:phases
       (modify-phases %standard-phases
         (add-after 'install 'patch-interpreter
           (lambda* (#:key inputs outputs #:allow-other-keys)
             (let* ((out    (assoc-ref outputs "out"))
                    (bin    (string-append out "/bin/<binary>"))
                    (glibc  (assoc-ref inputs "glibc"))
                    (gcc    (assoc-ref inputs "gcc"))
                    (linker (string-append glibc "/lib/ld-linux-x86-64.so.2"))
                    (rpath  (string-append glibc "/lib:" gcc "/lib")))
               (invoke "patchelf"
                       "--set-interpreter" linker
                       "--set-rpath" rpath
                       bin)))))))
    (synopsis "<short description>")
    (description "<longer description>")
    (home-page "<url>")
    (license license:expat)))
#+END_SRC

Key points:
- `#:validate-runpath? #f` — skip Guix's RUNPATH check
- `(list gcc "lib")` — the "lib" output contains libstdc++.so.6
- patchelf sets the interpreter to Guix's glibc linker
- rpath includes both glibc and gcc lib directories

*** Emacs package with dependencies

When an Emacs package requires other Emacs packages at compile time (has `(require 'some-dep)` at top level), you MUST list them as BOTH `inputs` AND `propagated-inputs`:

#+BEGIN_SRC scheme
(define-public emacs-<name>
  (package
    (name "emacs-<name>")
    (version "<version>")
    (source (origin
              (method git-fetch)
              (uri (git-reference
                    (url "https://github.com/<user>/<repo>")
                    (commit "<commit-hash>")))
              (file-name (git-file-name name version))
              (sha256
               (base32 "<hash-from-build-log>"))))
    (build-system emacs-build-system)
    (inputs (list emacs-<dep1> emacs-<dep2>))
    (propagated-inputs (list emacs-<dep1> emacs-<dep2>))
    (synopsis "<short description>")
    (description "<longer description>")
    (home-page "<url>")
    (license license:gpl3+)))
#+END_SRC

CRITICAL: `inputs` makes deps available during byte-compilation. `propagated-inputs` makes them available at runtime. You need BOTH when the package has `(require 'dep)` at the top level.

*** Git-fetch hash gotcha

The hash from `guix hash --recursive /path/to/checkout` will NOT match what Guix expects because:
1. Local checkout includes `.git/` directory
2. Guix's `git-fetch` excludes `.git/`

If you get a hash mismatch, the actual hash is in the build log:
#+BEGIN_SRC sh
zcat /var/log/guix/drvs/<XX>/<hash>.drv.gz | grep "actual hash"
#+END_SRC

Use the "actual hash" from the build log. It always starts with `0` or `1`.

*** Go packages with binary in subdirectory (cmd/<name> pattern)

Many Go projects have `go.mod` at the root but the main binary is in `cmd/<name>/`.
The `go-build-system` expects `#:import-path` to point to the module root (where
`go.mod` lives), but you must override the build phase to install the correct
subdirectory:

#+BEGIN_SRC scheme
(define-module (core-system user-space root <category> <name>)
  #:use-module (guix packages)
  #:use-module (guix download)
  #:use-module (guix git-download)
  #:use-module (guix build-system go)
  #:use-module (guix gexp)
  #:use-module (gnu packages golang)
  #:use-module (gnu packages golang-xyz)
  #:use-module ((guix licenses) #:prefix license:)
  #:export (<name>))

(define-public <name>
  (package
    (name "go-github-com-<user>-<repo>")
    (version "<version>")
    (source (origin
              (method git-fetch)
              (uri (git-reference
                    (url "https://github.com/<user>/<repo>")
                    (commit (string-append "v" version))))
              (file-name (git-file-name name version))
              (sha256
               (base32 "<hash>"))
              (modules '((guix build utils)))
              (snippet
               #~(begin
                   (substitute* "go.mod"
                     (("go 1.25.5") "go 1.24"))))))  ; patch Go version if needed
    (build-system go-build-system)
    (arguments
     (list
      #:import-path "github.com/<user>/<repo>"   ; module root, NOT cmd/
      #:install-source? #f
      #:phases
      #~(modify-phases %standard-phases
          (replace 'build
            (lambda* (#:key import-path #:allow-other-keys)
              (invoke "go" "install" "-ldflags=-s -w" "-trimpath"
                      (string-append import-path "/cmd/<binary>")))))))
    (propagated-inputs (list go-github-com-<dep1>
                             go-github-com-<dep2>))
    (synopsis "<short description>")
    (description "<longer description>")
    (home-page "https://github.com/<user>/<repo>")
    (license license:expat)))
#+END_SRC

CRITICAL: `#:import-path` must be the module root (where go.mod is), NOT the
cmd/ subdirectory. The custom build phase installs the specific binary.

To find Go dependency module paths:
#+BEGIN_SRC sh
guix import go github.com/<user>/<repo>
# Look at propagated-inputs in the output
#+END_SRC

To verify Go deps exist in Guix:
#+BEGIN_SRC scheme
(use-modules (gnu packages golang-xyz) (gnu packages golang-build))
;; Check each dep in both modules
#+END_SRC

*** npm FOD pattern (for large npm packages with many dependencies)

For npm packages with hundreds/thousands of dependencies (like n8n with 2443
packages), use a two-phase fixed-output derivation (FOD) pattern:

Phase 1: FOD for node_modules — downloads the npm tarball and runs
`npm install --ignore-scripts` to produce a reproducible node_modules snapshot.

Phase 2: Main package — unpacks the npm tarball (contains bin/, dist/, templates/)
and symlinks the FOD's node_modules into it.

#+BEGIN_SRC scheme
(define-module (core-system user-space root <CATEGORY> <NAME>)
  #:use-module (guix packages)
  #:use-module (guix download)
  #:use-module (guix build-system trivial)
  #:use-module (guix gexp)
  #:use-module (gnu packages node)
  #:use-module (gnu packages base)
  #:use-module (gnu packages compression)
  #:use-module ((guix licenses) #:prefix license:)
  #:export (<NAME>))

;; Phase 1: FOD for node_modules
(define <NAME>-node-modules
  (package
    (name "<NAME>-node-modules")
    (version "<VERSION>")
    (source (origin
              (method url-fetch)
              (uri "https://registry.npmjs.org/<NAME>/-/<NAME>-<VERSION>.tgz")
              (sha256 (base32 "<TARBALL-HASH>"))))
    (build-system trivial-build-system)
    (arguments
     (list
      #:modules '((guix build utils))
      #:builder
      #~(begin
          (use-modules (guix build utils))
          (let* ((node (assoc-ref %build-inputs "node"))
                 (npm  (string-append node "/bin/npm"))
                 (tar  (string-append (assoc-ref %build-inputs "tar") "/bin/tar"))
                 (gzip (string-append (assoc-ref %build-inputs "gzip") "/bin/gzip"))
                 (src  (assoc-ref %build-inputs "source"))
                 (out  (assoc-ref %outputs "out"))
                 (tmp  (string-append out "/.npm-tmp"))
                 (work "/tmp/work"))
            (setenv "HOME" out)
            (setenv "npm_config_cache" tmp)
            (setenv "npm_config_tmp" tmp)
            (mkdir-p out)
            (mkdir-p tmp)
            (mkdir-p work)
            (invoke tar "--use-compress-program" gzip "-xf" src "-C" work)
            (with-directory-excursion (string-append work "/package")
              (invoke npm "install"
                      "--ignore-scripts"
                      "--no-audit"
                      "--no-fund"
                      "--prefix" out))))))
    (inputs (list node tar gzip))
    (outputs '("out"))
    (home-page "<URL>")
    (synopsis "node_modules for <NAME>")
    (description "FOD npm dependency snapshot for <NAME>.")
    (license license:expat)))

;; Phase 2: Main package
(define-public <NAME>
  (package
    (name "<NAME>")
    (version "<VERSION>")
    (source (origin
              (method url-fetch)
              (uri "https://registry.npmjs.org/<NAME>/-/<NAME>-<VERSION>.tgz")
              (sha256 (base32 "<TARBALL-HASH>"))))
    (build-system trivial-build-system)
    (arguments
     (list
      #:modules '((guix build utils))
      #:builder
      #~(begin
          (use-modules (guix build utils))
          (let* ((out   #$output)
                 (bin   (string-append out "/bin"))
                 (lib   (string-append out "/lib/node_modules/<NAME>"))
                 (tar   (string-append #$tar "/bin/tar"))
                 (gzip  (string-append #$gzip "/bin/gzip"))
                 (nm-src (string-append #$(package-file <NAME>-node-modules) "/lib/node_modules/<NAME>/node_modules"))
                 (node  (string-append #$node "/bin/node")))
            (mkdir-p lib)
            (mkdir-p bin)
            (invoke tar "--use-compress-program" gzip "-xf" #$source "--strip-components=1" "-C" lib)
            (symlink nm-src (string-append lib "/node_modules"))
            (call-with-output-file (string-append bin "/NAME>")
              (lambda (p)
                (format p "#!/bin/sh\nexec ~a ~a/bin/<NAME> \"$@\"\n"
                        node lib)))
            (chmod (string-append bin "/NAME>") #o755)))))
    (inputs (list node tar gzip <NAME>-node-modules))
    (outputs '("out"))
    (home-page "<URL>")
    (synopsis "<SHORT DESCRIPTION>")
    (description "<LONGER DESCRIPTION>")
    (license license:expat)))
#+END_SRC

Key points:
- Get tarball hash: `guix download https://registry.npmjs.org/<NAME>/-/<NAME>-<VERSION>.tgz`
- The FOD's hash is derived from the npm tarball hash — same source, same hash
- `--ignore-scripts` prevents arbitrary code execution during npm install (required for Guix reproducibility)
- `--no-audit --no-fund` speeds up install
- Set HOME and npm_config_cache/tmp to avoid /homeless-shelter errors
- Use `tar --use-compress-program gzip` because tar needs gzip in PATH
- Wrapper script must point to the bin/<NAME> entry point, not dist/index.js

** Template E — Container module (current flat pattern)

#+BEGIN_SRC scheme
(define-module (systems containers <NAME>)
  #:use-module (gnu system)
  #:use-module (gnu services)
  #:use-module (gnu services docker)
  #:use-module (systems containers core-container)
  #:export (<NAME>-os <NAME>-service))

(define-public <NAME>-os
  (operating-system
    (inherit container-os)
    (host-name "guixos-<NAME>")
    (packages (append (list <PKG>)
                      (operating-system-packages container-os)))))

(define-public <NAME>-service
  (service oci-container-service-type
           (list (oci-container-configuration
                   (image (oci-image
                            (repository "guixos-<NAME>")
                            (tag "latest")
                            (value <NAME>-os)))
                   (network "host")
                   (auto-start? #t)
                   (respawn? #t)))))
#+END_SRC
